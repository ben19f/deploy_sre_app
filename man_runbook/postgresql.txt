#### создаем скрипт scriptpg.sh  

``` sql
#!/bin/bash
set -e

DB=sre
USER=sre

psql -v ON_ERROR_STOP=1 -U "$USER" -d "$DB" -h 127.0.0.1 -c "
CREATE TABLE IF NOT EXISTS author (
  id SERIAL,
  name VARCHAR(100)
);
"

psql -v ON_ERROR_STOP=1 -U "$USER" -d "$DB" -h 127.0.0.1 -c "
INSERT INTO author (name) VALUES ('John'), ('Alex');
"
```
